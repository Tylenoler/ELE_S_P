#include <Adafruit_NeoPixel.h>

#define PIN 6

// 原始帧数据
word ledarray[] = {
    // ... 原始数据 ...
};

// 添加新帧数据
word ledarray_frame1[] = {
    // ... 第一帧数据 ...
};

word ledarray_frame2[] = {
    // ... 第二帧数据 ...
};

word ledarray_frame3[] = {
    // ... 第三帧数据 ...
};

// 帧指针数组
word* frames[] = {
    ledarray,       // 原始帧
    ledarray_frame1,
    ledarray_frame2,
    ledarray_frame3
};

int frameCount = sizeof(frames) / sizeof(frames[0]); // 自动计算帧数
int currentFrame = 0;

Adafruit_NeoPixel strip = Adafruit_NeoPixel(256, PIN, NEO_GRB + NEO_KHZ800);

// 帧率控制变量
unsigned long previousMillis = 0;
const long interval = 100; // 帧间隔(毫秒)，100ms = 10FPS

void setup()
{
    strip.begin();
    strip.show();   // 初始化所有LED为关闭状态
}

void loop()
{
    unsigned long currentMillis = millis();
    
    // 检查是否到达下一帧时间
    if (currentMillis - previousMillis >= interval) {
        previousMillis = currentMillis;
        
        // 渲染当前帧
        RenderFrame(currentFrame);
        
        // 前进到下一帧
        currentFrame = (currentFrame + 1) % frameCount;
    }
}

void RenderFrame(int frameIndex)
{
    word* currentFrameData = frames[frameIndex];
    
    for (int t = 0; t < 256; t++)
    {
      strip.setPixelColor(t, currentFrameData[t]); 
    }
  
    strip.show();
}
